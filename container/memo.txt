- commands for docker
  - Login to the ksql-cli container
    - $ sudo docker exec -it ksql-cli /bin/bash
    - $ sudo docker exec -it streams /bin/bash
  - delete all images
    - $ sudo docker rmi $(sudo docker images -q)
  - build images which created by docker-compose.yml
    - $ sudo docker-compose build
    - $ sudo docker-compose up -d
  - show a log in each container
    - $ sudo docker-compose logs streams
